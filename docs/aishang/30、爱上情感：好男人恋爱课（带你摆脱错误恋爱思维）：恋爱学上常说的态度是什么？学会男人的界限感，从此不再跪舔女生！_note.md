    def __init__(self, self_accept, self_not_accept, others_accept, others_not_accept):
        self.self_accept = self_accept
        self.self_not_accept = self_not_accept
        self.others_accept = others_accept
        self.others_not_accept = others_not_accept

        print("对方可以接受的行为:", self.others_accept)
