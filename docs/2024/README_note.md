    n = len(arr)
    for i in range(n):
        min_idx = i
        for j in range(i+1, n):

    low = 0
    high = len(arr) - 1
    mid = 0

    while low <= high:
        mid = (high + low) // 2

        if arr[mid] < x:
            low = mid + 1
