# 定义一个函数，用于检查一个人是否爱自己
def is_loving_yourself(person):
    # 检查这个人是否购买新衣服、鞋子、包包，享受美食和娱乐
    if person.buys_new_clothes and person.enjoys_foods and person.has_fun:
        return True
    else:
        return False

# 创建一个示例人物
person = {
    'buys_new_clothes': True,
    'enjoys_foods': True,
    'has_fun': True
}

# 检查这个人是否爱自己
print(is_loving_yourself(person))
